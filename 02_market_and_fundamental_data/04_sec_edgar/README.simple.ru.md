–°–æ–∑–¥–∞–¥–∏–º —Å–∫—Ä–∏–Ω–µ—Ä, –∫–æ—Ç–æ—Ä—ã–π –Ω–∞–π–¥–µ—Ç –∫–æ–º–ø–∞–Ω–∏–∏ —Å —Ö–æ—Ä–æ—à–∏–º–∏ —Ñ–∏–Ω–∞–Ω—Å–æ–≤—ã–º–∏ –ø–æ–∫–∞–∑–∞—Ç–µ–ª—è–º–∏:

```python
import pandas as pd
import requests
from io import BytesIO
from zipfile import ZipFile

# –ö—Ä–∏—Ç–µ—Ä–∏–∏ –æ—Ç–±–æ—Ä–∞
MIN_REVENUE = 1e9  # –ú–∏–Ω–∏–º—É–º $1 –º–ª—Ä–¥ –≤—ã—Ä—É—á–∫–∏
MIN_PROFIT_MARGIN = 0.15  # –ú–∏–Ω–∏–º—É–º 15% —Ä–µ–Ω—Ç–∞–±–µ–ª—å–Ω–æ—Å—Ç—å
MAX_DEBT_TO_EQUITY = 0.5  # –î–æ–ª–≥ –Ω–µ –±–æ–ª—å—à–µ 50% –æ—Ç –∫–∞–ø–∏—Ç–∞–ª–∞

# –°–∫–∞—á–∞—Ç—å –ø–æ—Å–ª–µ–¥–Ω–∏–π –∫–≤–∞—Ä—Ç–∞–ª
url = "https://www.sec.gov/files/dera/data/financial-statement-and-notes-data-sets/2023q4.zip"
response = requests.get(url)
zip_file = ZipFile(BytesIO(response.content))

sub = pd.read_csv(zip_file.open('sub.txt'), sep='\t', encoding='latin1')
num = pd.read_csv(zip_file.open('num.txt'), sep='\t', encoding='latin1')

# –¢–æ–ª—å–∫–æ –≥–æ–¥–æ–≤—ã–µ –æ—Ç—á–µ—Ç—ã
annual_reports = sub[sub['form'] == '10-K']

results = []

for idx, report in annual_reports.iterrows():
    adsh = report['adsh']
    company = report['name']

    # –ò–∑–≤–ª–µ—á—å –¥–∞–Ω–Ω—ã–µ
    company_data = num[num['adsh'] == adsh]

    # –ü–æ–ª—É—á–∏—Ç—å –º–µ—Ç—Ä–∏–∫–∏
    revenue = company_data[company_data['tag'] == 'Revenues']
    net_income = company_data[company_data['tag'] == 'NetIncomeLoss']
    assets = company_data[company_data['tag'] == 'Assets']
    liabilities = company_data[company_data['tag'] == 'Liabilities']

    # –ü—Ä–æ–≤–µ—Ä–∏—Ç—å —á—Ç–æ –≤—Å–µ –¥–∞–Ω–Ω—ã–µ –µ—Å—Ç—å
    if (len(revenue) > 0 and len(net_income) > 0 and
        len(assets) > 0 and len(liabilities) > 0):

        rev = revenue['value'].iloc[0]
        ni = net_income['value'].iloc[0]
        asset = assets['value'].iloc[0]
        liab = liabilities['value'].iloc[0]

        # –í—ã—á–∏—Å–ª–∏—Ç—å –ø–æ–∫–∞–∑–∞—Ç–µ–ª–∏
        profit_margin = ni / rev if rev > 0 else 0
        equity = asset - liab
        debt_to_equity = liab / equity if equity > 0 else 999

        # –ü—Ä–∏–º–µ–Ω–∏—Ç—å —Ñ–∏–ª—å—Ç—Ä—ã
        if (rev >= MIN_REVENUE and
            profit_margin >= MIN_PROFIT_MARGIN and
            debt_to_equity <= MAX_DEBT_TO_EQUITY):

            results.append({
                'Company': company,
                'Revenue': rev,
                'Net Income': ni,
                'Profit Margin': profit_margin,
                'Debt/Equity': debt_to_equity
            })

# –°–æ–∑–¥–∞—Ç—å DataFrame —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
screener = pd.DataFrame(results)
screener = screener.sort_values('Profit Margin', ascending=False)

for idx, row in screener.head(10).iterrows():
    print(f"{row['Company']}")

    print(f"  –ß–∏—Å—Ç–∞—è –ø—Ä–∏–±—ã–ª—å: ${row['Net Income']/1e9:.1f}B")
    print(f"  –†–µ–Ω—Ç–∞–±–µ–ª—å–Ω–æ—Å—Ç—å: {row['Profit Margin']:.1%}")
    print(f"  –î–æ–ª–≥/–ö–∞–ø–∏—Ç–∞–ª: {row['Debt/Equity']:.2f}")
    print()

# –í–æ–∑–º–æ–∂–Ω—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç:
```
