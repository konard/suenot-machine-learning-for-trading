# 6. RSI
def calculate_rsi(prices, period=14):
    delta = prices.diff()
    gain = (delta.where(delta > 0, 0)).rolling(period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(period).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

rsi = calculate_rsi(prices)

# 7. MACD (Moving Average Convergence Divergence)

def handle_data(context, data):
    # –í—ã—á–∏—Å–ª—è–µ–º –Ω–∞—à —Ñ–∞–∫—Ç–æ—Ä
    prices = data.history(context.stocks, 'price', 60, '1d')
    momentum = prices.pct_change(60)
    volatility = prices.pct_change().rolling(60).std()
    factor = (momentum / volatility).iloc[-1]

    # –í—ã–±–∏—Ä–∞–µ–º —Ç–æ–ø-10 –ø–æ —Ñ–∞–∫—Ç–æ—Ä—É
    top_10 = factor.nlargest(10)

    # –ü–æ–∫—É–ø–∞–µ–º –∏—Ö
    for stock in top_10.index:
        order_target_percent(stock, 0.10)  # –ü–æ 10% –Ω–∞ –∫–∞–∂–¥—É—é

    # –ü—Ä–æ–¥–∞—ë–º –æ—Å—Ç–∞–ª—å–Ω—ã–µ
    for stock in context.portfolio.positions:
        if stock not in top_10:
            order_target_percent(stock, 0)
